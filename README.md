## Install

```bash
npm install --save basicformhook
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'basicformhook'
import 'basicformhook/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
